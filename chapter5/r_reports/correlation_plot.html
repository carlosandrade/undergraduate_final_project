Derby, Lucene, Pdfbox, Ivy and Ftpserver are all projects of the Apache Software Foundation. Some data pertaining each of these projects is loaded on each variable.

The purpose of this analysis is to observe if we can establish any relationship between structural complexity of code files and the amount of effort that was taken to maintain them. Concretly, we operationalize structural complexity as a set of OO Metrics, each of which measure structural complexity. Effort is operationalized on the variables discussion, actions and churn.

Each observation is given in a row and can be considered as follows: For every change to a file to address an issue in a given release we calculate structural complexity file metrics and the associated effort. Concretely, each row is identified by the columns file, issue_code and release. A discussion in respect of how each effort estimator is mapped to a file metric is discussed on the paper.

Since this is a cross-sectional study, we must train and test our models in a given point in time. Since we only take measures per release, we further consider the set of datapoints that belong to each release as potential training or test sets. We must be careful however to analyze which releases can be used according to their size (some may lack enough data points to be used).

filterReleases <- function(data, threshold) {
    # Obtain the release values that fall below the threshold for this dataset
    data.perRelease = amountDataPointsPerRelease(data)
    releases = data.perRelease[data.perRelease$n > threshold, 1]
    # Return datapoints that belongs only to the releases above the threshold
    data = data[data$release %in% releases, ]
    data
}

Define a function that calculates the correlation given a method, rho and significance thresholds. It will also only print the upper triangle (since the lower is the same and the diagonal is always 1)

calc_corr <- function(dataframe, rho_threshold, significance, cor_method) {
    # This function fills with NA all values that does not match the significance and rho values criteria.
    suppressMessages(library(Hmisc))
    cor_table = rcorr(as.matrix(dataframe), type = cor_method)

    # lower_triangle = lower.tri(cor_table[[1]], diag = TRUE) This function prints both upper and lower triangle.

    thresholds = (cor_table[[3]] > significance) | (cor_table[[1]] < rho_threshold)
    cor_table[[1]][thresholds] = NA
    cor_table[[1]]
    # cor_table[[1]][1:3,1:ncol(cor_table[[1]])]
}

Create a 3 element list that contains the pair of correlation metrics, its rho and release value.

list_correlations <- function(project.all.list, rho, p, cor_method) {
    # This calculates the list correlations of cost estimators versus themselves, and cost estimators versus file metrics. In other words, it lists all
    # correlations but those among the file metrics.


    # Preallocate a huge dataframe to avoid dynamic allocation
    N <- 10000
    project.correlation = data.frame(rho = rep(NA, N), v1 = rep(NA, N), v2 = rep(NA, N), release = rep(NA, N), stringsAsFactors = FALSE)
    project.correlation$v1 = as.character(project.correlation$v1)
    project.correlation$v2 = as.character(project.correlation$v2)
    project.correlation$release = as.character(project.correlation$release)

    counter = 0
    for (i in 1:length(project.all.list)) {
        cor_table = calc_corr(project.all.list[[i]], rho, p, cor_method)
        for (idx in 1:3) {
            for (jdx in 1:ncol(cor_table)) {
                if ((idx != jdx) && !(idx >= 2 && jdx <= 3) && !is.na(cor_table[idx, jdx])) {
                  #idx >=2 && jdx <=3 avoids e.g churn - actions ; actions - churn pairs
                  counter = counter + 1
                  newrow = c(rho = cor_table[idx, jdx], v1 = rownames(cor_table)[idx], v2 = colnames(cor_table)[jdx], release = i)
                  # project.correlation = rbind(project.correlation,newrow, deparse.level = 1)
                  project.correlation[counter, ] = newrow
                }
            }
        }
    }
    project.correlation = project.correlation[complete.cases(project.correlation), ]
    project.correlation
}

Plot the correlations


plot_correlation <- function(project.list.correlation,project_name){
# Function adapted from http://solomonmessing.wordpress.com/2012/03/04/visualization-series-insight-from-cleveland-and-tufte-on-plotting-numeric-data-by-groups/  


  #Merge the columns v1 and v2 in a single column named correlation pairs
  concatenated_pairs = paste0(project.list.correlation[,2],' - ',project.list.correlation[,3])
  project.list.correlation[,2] = concatenated_pairs
  project.list.correlation = project.list.correlation[,c(1,2,4)] 
  colnames(project.list.correlation)[2] = "correlation_pairs"

  #adjust rho to be numeric
  project.list.correlation$rho = as.numeric(project.list.correlation$rho)

 suppressMessages(library(plyr))


  # first we use plyr to calculate the mean rating and SE for each story
  ratingdat <- ddply(project.list.correlation, c("correlation_pairs"), summarise,
        M = mean(rho, na.rm=T),
        SE = sd(rho, na.rm=T),#/sqrt(length(na.omit(rho))),
        N = length(na.omit(rho))) #N will indicate the amount of releases a given pair appear in a project in this case

 # Replace sd values of rho from NA to 0. This only occurs when there is only a single value reported for a correlation pair in all releases. 
  ratingdat[is.na(ratingdat$SE),"SE"] = 0  

  #Add the amount of releases the correlation occurs with significance level p.
  ratingdat[,1] = paste0(ratingdat[,1],' | ',ratingdat[,4])  

 # make correlation pairs into an ordered factor, ordering by mean rho of the pair across the project releases:
  ratingdat$correlation_pairs <- factor(ratingdat$correlation_pairs)
  ratingdat$correlation_pairs <- reorder(ratingdat$correlation_pairs, ratingdat$M)



  # Now open a connection to a pdf file
 # if(!is.na(directory) && !is.na(filename)){
#    setwd(directory)
    #pdf(file=filename, height=14, width=8.5)
 # }

    title = paste(project_name,"Spearman Correlations, with SD across",nlevels(factor((project.list.correlation$release))),"releases")
    suppressMessages(library(ggplot2))
    ggplot(ratingdat, aes(x = M, xmin = M-SE, xmax = M+SE, y = correlation_pairs )) +
          geom_point() + geom_segment( aes(x = M-SE, xend = M+SE,
                         y = correlation_pairs, yend=correlation_pairs)) +
          theme_bw() + theme(axis.title.x = element_text(size = 12, vjust = .25))+
         xlab("Mean rho") + ylab("Correlation Pair | N. Releases p < 0.01") +
         labs(title = title)

    #if(!is.na(directory) && !is.na(filename)){
      #dev.off()
  #  }
  #}
}
